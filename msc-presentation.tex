\begin{frame}
\frametitle{Sistemas Amostrados Lineares}


\begin{align}
\dot{x}(t) &= Ax(t) + Bu(t) + Gw(t) \notag \\ 
y(t_k) &= Cx(t_k) + v(t_k) \notag
\end{align}

\vspace{1cm}
\noindent
Discretizado por:


\begin{equation}
x(t_{k+1}) = A_\text{d}(t_k,t_{k+1}) x(t_k) + B_\text{d}(t_k,t_{k+1}) u(t_k) + w_\text{d}(t_k,t_{k+1}) \notag
\end{equation}
\vfill
\pause
\begin{block}{}
	\centering
Método de estimação: \textbf{Filtro de Kalman}
\end{block}


\end{frame}
%------------------------------------------------

\begin{frame}
\frametitle{Sistemas Amostrados Não Lineares}

\begin{align}
\dot{x}(t) &= f(x(t), u(t), w(t), t) \notag \\ 
y(t_k) &= g(x(t_k) ,v(t_k), t_k) \notag
\end{align}

\vspace{1cm}
\noindent 
Discretizado por Runge-Kutta:

\begin{align}
x(t_{k+1})&= x({t_k}) + \frac{1}{6} \left( k_1 + 2k_2 +  3k_3 + k_4\right), \notag\\
t_{k+1} &= t_k + h_k, \label{eq:time_interval} \notag
\end{align}

\vfill
\pause
\begin{block}{}
	\centering
	Método de estimação: \textbf{Filtro de Kalman \textit{Unscented}}
\end{block}


\end{frame}

\begin{frame}
\frametitle{Formulação do Problema}

Queremos estimar o vetor de estados $x(iT)$ e sua covariância de forma recursiva, em intervalos igualmente espaçados $T$, considerando:
\vspace{0.5cm}
\begin{itemize}
	\item Instantes de tempo das observações $t_k$ é definido pelo intervalo:
	\begin{itemize}
		\item[$\rightarrow$] $h_k \triangleq t_k - t_{k-1}$
		\item[$\rightarrow$] $h_k \sim \mathcal{E}(\lambda)$
	\end{itemize}
\vspace{0.35cm}
	\item Entrada $u(t)$ é atualizada em intervalos de tempo constantes $T$:
	\begin{itemize}
		\item[$\rightarrow$] $u(t) = u(iT), \text{ para } iT \leq t < (i+1)T$
		\item[$\rightarrow$] $i = 0, \ 1, \ 2, ... \in \mathbb{N}$
	\end{itemize} 
\end{itemize}

\end{frame}

\item Modelo de medição $\rightarrow$ $y(t_k) = g(x(t_k),v(t_k),t_k)$\\
\item Deslocamento de tempo $\rightarrow$ \textbf{$\delta_k \triangleq nT - t_k$},
\end{itemize}

e_k &= g(t_k) - g(t_k + \delta_k),  \\
e_k & = \left[ \frac{g(t_k) - g(t_k + \delta_k)}{\delta_k} \right] \delta_k  \\
e_k & \approx -\frac{dy}{dt}\delta_k.  
\end{align*}

\begin{frame}
\frametitle{Erro introduzido pelos deslocamentos}
\centering

\begin{itemize}
\item Sinal verdadeiro $\rightarrow$ $y(t)$
\item Medições $\rightarrow$ $\hat{y}(t_k)$
\item Medições deslocadas $\rightarrow$ $\tilde{y}(t_k + \delta_k) = \hat{y}(t_k) $
\end{itemize}


\pause
\begin{align*}
\text{RMSE}_{\hat{y}} & \triangleq \sqrt{\frac{\sum_{k=1}^{N} (\hat{y}(t_k) - y(t_k))}{N}} \\
\text{RMSE}_{\tilde{y}} & \triangleq \sqrt{\frac{\sum_{k=1}^{N} (\tilde{y}(t_k+\delta_k) - y(t_k+\delta_k))}{N}} \\
\end{align*}

\pause
\begin{block}{Contribuição dos deslocamentos para o RMSE total}
\centering
$\text{RMSE}_{e} \triangleq \text{RMSE}_{\tilde{y}} - \text{RMSE}_{\hat{y}}$

\end{block}


\end{frame}
%------------------------------------------------

\begin{frame}
\frametitle{Em função do ruído e da frequência média de amostragem}
\centering

$\text{RMSE}_{e} \triangleq \text{RMSE}_{\tilde{y}} - \text{RMSE}_{\hat{y}}$

%------------------------------------------------
\begin{frame}
\frametitle{Precisão}

Raiz do Erro Quadrático Médio:

\begin{equation*}
\text{RMSE} = \sqrt{ \frac{ \sum_{k=1}^{N} (\hat{x}_{k|k} -x_k)^2}{N}}
\end{equation*}
\vspace{0.2cm}
Testes estatísticos
\begin{equation*}
\mu_{\text{D}} = RMSE_\text{w/o} - RMSE_\text{w}
\end{equation*}
\begin{equation*}
\begin{cases}
H_0: \mu_{\textrm{D}} &= 0, \\
H_1: \mu_{\textrm{D}} &\neq 0,
\end{cases}
\end{equation*}
%
Tamanho de efeito:

\begin{equation*}
d = \frac{\overline{RMSE}_{\text{w}} - \overline{RMSE}_{\text{w/o}}}{s_\text{D}}
\end{equation*}


\end{frame}

%------------------------------------------------
\begin{frame}
\frametitle{Consistência}
Um estimador é dito \textbf{consistente} se:
\begin{align*}
E\left[x_k - \hat{x}_{k|k}\right] &\triangleq E[\tilde{x}_{k|k}] = 0 \\
E\left[(x_k - \hat{x}_{k|k})(x_k - \hat{x}_{k|k})^T\right] &\triangleq E[\tilde{x}_{k|k}\tilde{x}_{k|k}^t] = P^{xx}_{k|k}
\end{align*}

Uma forma de testar é definindo:
\begin{align*}
NEES_k &\triangleq  \tilde{x}_{k|k}^T(P^{xx}_{k|k})^{-1}\tilde{x}_{k|k} \\
NIS_k &\triangleq  \eta_{k|k-1}^T(P^{yy}_{k|k-1})^{-1}\eta_{k|k-1} \\
\end{align*}

\vfill
Teste de hipóteses: $H_0 : NEES_k \sim \chi^2(n_x), \ NIS_k \sim \chi^2(n_y)$

\end{frame}

%------------------------------------------------
\begin{frame}
\frametitle{Consistência}
\centering
$P \left\{ NEES_k, NIS_k \in [r_1,r_2] | H_0 \right\} = 1 - \alpha$

Considere o sistema de um robô móvel não-holonômico:

\begin{minipage}{.70\textwidth}

	\begin{split}
	\dot{p}_{\textrm{x}} & = v\cos (\theta),\\
	\dot{p}_{\textrm{y}} & = v\sin (\theta),\\
	\dot{\theta}  & = u_1(t),\\
	\dot{v} & = u_2(t),
	\end{split}
	\end{equation*}

	\vspace{0.2cm}

		\hfill
		$v$:								& velocidade linear, \\
		\hfill
		$u_1$:	 							& entrada: velocidade angular ($\omega$), \\
		\hfill
		$u_2$:	 							& entrada: aceleração linear ($a$)
	\end{tabular}

\begin{equation*}
u_i = [\omega_i\ a_i]^T,

%Observations are taken at random time instants $t_k$ and are considered to be a strictly increasing sequence ($t_{k+1}>t_k,\forall k \in \mathbb{N^+}$) and defined by the time intervals $h_k \triangleq t_k-t_{k-1}, \ \forall k \geq 1$. In this work, we assume that the observation time instants $t_k$ are given by a Poisson random process. In other words, the time intervals $h_k$ are independent and identically distributed (i.i.d.) exponential random variables (RVs) with a known rate parameter $\lambda$, that is $h_k \sim \mathcal{E} (\lambda)$. Since the expected value of an exponential RV is the inverse of its rate parameter, $\lambda$ will sometimes be referred to as expected or average sampling rate of the irregular sampled quantity. An example of time intervals produced by such a random process is illustrated in Figure \ref{fig:amost}. This sampling model characterizes a common application for an event-based sampling scheme or for a networked control system with unsynchronized sensors. \citep{Micheli2002} considered a set of $N$ identical sensors measuring the state variables of a physical process every $L$ seconds. They have shown that, if the sensors are independent and unsynchronized and $N$ is large enough, then the waiting time between the realization of two consecutive measurements can be approximated by an exponential random variable $\mathcal{E} (\lambda)$, where the parameter is given by $\lambda = N/L$.
