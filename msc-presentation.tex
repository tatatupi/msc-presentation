\subsection{Desafios}

\begin{frame}
\frametitle{Desafios}

\begin{block}{}
	Falta de sincronização entre os múltiplos sensores da rede pode levar a \textbf{amostragem irregular} sem informação confiável de \textbf{carimbo de tempo}
\end{block}

\vspace{1cm}

\textbf{Soluções}:

\begin{itemize}
	\item Investir em sincronização
	\item Deslocar os instantes de tempo
\end{itemize}

\end{frame}

\subsection{Objetivos}
\begin{frame}

\frametitle{Objetivos}


\begin{enumerate}
	\item Revisar os métodos de fusão sensorial e o problema de amostragem irregular;
	\item Discutir os algoritmos e suas adaptaçãos ao modelo de medição amostrado irregularmente, sem carimbo de tempo;
	\item Desenvolver uma metodologia para estudar os efeitos de desconsiderar os carimbos de tempo;
	\item Aplicar a metodologia em um sistema linear e outro não-linear, utilizando índices de desempenho que avaliam a precisão e a consistência de estimação;
\end{enumerate}

\end{frame}

\subsection{Modelo de Amostragem Irregular}
%------------------------------------------------

\begin{frame}
\frametitle{Modelo de Amostragem Irregular}

\begin{block}{}
Instantes de amostragem modelados por um processo de Poisson:
\end{block}


\begin{align}
P\left(N(t)=n\right) &= e^{-\lambda t}\frac{(\lambda t)^n}{n!} \notag\\ \notag\\
\rho_{h_k}(t) &= \lambda e^{-\lambda t}\notag
\end{align}
\end{frame}
